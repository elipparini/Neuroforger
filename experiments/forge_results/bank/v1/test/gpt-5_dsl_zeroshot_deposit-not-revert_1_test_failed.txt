pragma solidity ^0.8.13;
import "forge-std/Test.sol";
import "../Bank.sol";

contract Dummy {}

contract BankTest is Test {     
    Bank immutable bank;
    
    constructor() {
        // deploying a Bank contract
        address bank_deployer = address(0xB0B);
        vm.prank(bank_deployer);
        bank = new Bank();
    }

    function test_not_deposit_revert_violation() public {
        // Ensure the pranked user has zero balance so sending value will fail
        vm.deal(address(0xABCD), 0);

        address user = address(0xABCD);
        
        vm.prank(user);
        uint256 msg_value = 1 wei;

        vm.expectRevert();
        bank.deposit{value: msg_value}();
    }
}
